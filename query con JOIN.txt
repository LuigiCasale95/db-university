-- 1. Selezionare tutti gli studenti iscritti al Corso di Laurea in Economia

SELECT *
FROM students
JOIN degrees 
ON department_id = degrees.department_id
WHERE degrees.name = 'Corso di Laurea in Economia' ;

-- 2. Selezionare tutti i Corsi di Laurea Magistrale del Dipartimento di Neuroscienze

SELECT department_id, departments.name, degrees.name, degrees.level
FROM  departments
JOIN degrees 
ON degrees.level = 'magistrale'
WHERE departments.name = 'Dipartimento di Neuroscienze';

-- 3. Selezionare tutti i corsi in cui insegna Fulvio Amato (id=44)

SELECT course_teacher.teacher_id AS teacher, teachers.name, teachers.surname, courses.name AS corso, courses.description
FROM teachers
JOIN course_teacher
JOIN courses ON (teachers.id = course_teacher.teacher_id and courses.id = course_teacher.course_id)
WHERE teachers.id = 44;

-- 4. Selezionare tutti gli studenti con i dati relativi al corso di laurea a cui sono iscritti e il relativo dipartimento, in ordine alfabetico per cognome e nome

SELECT students.name, students.surname, departments.name AS Nome_Dipartimento, degrees.name AS Nome_corso
FROM students
JOIN degrees ON students.degree_id = degrees.id
JOIN departments ON degrees.department_id = departments.id;
 

-- 5. Selezionare tutti i corsi di laurea con i relativi corsi e insegnanti

 SELECT courses.name, teachers.name, teachers.surname
 FROM teachers
 JOIN course_teacher ON course_teacher.teacher_id = teacher_id
 JOIN courses ON  course_id = course_teacher.course_id
 join degrees on degrees.id = courses.degree_id;

-- 6. Selezionare tutti i docenti che insegnano nel Dipartimento di Matematica (54)

SELECT distinct teachers.name, teachers.surname, departments.name
FROM teachers
JOIN course_teacher ON course_teacher.teacher_id = teachers.id
JOIN courses ON course_id = course_teacher.course_id
JOIN degrees ON degrees.id = courses.degree_id
JOIN departments ON departments.id  = degrees.department_id
WHERE departments.name = 'Dipartimento di Matematica';
